 58def read_spaceZtxt(fname):
 59    with open(fname, 'r') as fin:
 60        line = fin.readline().split()
 61        sx = int(line[0])  # read number of x points
 62        sy = int(line[1])  # read number of y points
 63        print(f'Pixels: {sx} x {sy}')
 64
 65        spacex = float(line[2])  # read x spacing
 66        spacey = float(line[3])  # read y spacing
 67
 68    plu = np.loadtxt(fname, usecols=range(sy), skiprows=1)
 69    plu = (plu - np.mean(plu)) * (10 ** 6)  # 10^6 from mm to nm
 70
 71    rangeX = sx * spacex
 72    rangeY = sy * spacey
 73    x = np.linspace(0, sx * spacex, num=sx)
 74    y = np.linspace(0, sy * spacey, num=sy)
 75
 76    # create main XYZ and backup of original points in Z0
 77    X, Y = np.meshgrid(self.x, self.y)
 78    Z = np.transpose(plu)
 79
 80    return X, Y, Z, x, y

 82def read_xyztxt(fname):
 83    X, Y, Z = np.genfromtxt(fname, unpack=True, usecols=(0, 1, 2), delimiter=';')
 84
 85    # find size of array
 86    i = np.argwhere(Y > 0)[0][0]
 87
 88    X = np.reshape(X, (i, i))
 89    Y = np.reshape(Y, (i, i))
 90    Z = np.reshape(Z, (i, i))
 91
 92    x = X[0,:]
 93    y = Y[:,0]
 94
 95    return X, Y, Z, x, y

100def read_lextinfo(filecontent):
101    """read factor to convert raw data values into values in micron
102
103    parameter:
104    filecontent: binary content of complete file
105
106    return:
107    nx, ny: number of data points/ pixel for each direction
108    endheightmap: byte position of end of height map
109    heightperpix: factor to be multiplied to raw data to obtain micron
110    magnification: of objective lens
111    measdate: date an time when data where aquired with instrument
112    """
113    keywords = [b'HeightDataPerPixel', b'HEIGHT', \
114                b'ObjectiveLenseMagnification', b'LEXT 1.0.2.1', \
115                b'ImageWidth', b'ImageHeight', \
116                b'MakerCalibrationValue', b'HeightDataUnitZ', \
117                b'HeightDataUnitX', b'HeightDataUnitY', \
118                b'HeightMaxValue', b'HeightMaxScalingValue']
119    axes = [b'X', b'Y', b'Z']
120    #
121    #   according to format description of Olympus, HeightperPixel[X,Y] is
122    #   always in Picometer
123    #    print('filecontent: ', filecontent)
124    infostring7 = extract_tag(filecontent, keywords[7])
125    infostring8 = extract_tag(filecontent, keywords[8])
126    infostring9 = extract_tag(filecontent, keywords[9])
127    print('unit  z: ', infostring7)
128    print('unit  x: ', infostring8)
129    print('unit  y: ', infostring9)
130    tomicron = np.array([(0.001 ** int(infostring8)), (0.001 ** int(infostring9)), \
131                         (0.001 ** int(infostring7))])
132    infostring10 = extract_tag(filecontent, keywords[10])
133    infostring11 = extract_tag(filecontent, keywords[11])
134    print('HeightMaxValue: ', int(infostring10), ', HeightMaxScalingValue: ', int(infostring11))
135    if int(infostring10) * int(infostring11) == 0:
136        print('attention - this is interpreted not to have the correct HeightDataUnit')
137        print('from conversion of pior to lext format, therefore correction from mm to nm')
138        print('finally delivered in micron, as usual')
139        tomicron *= 1e-6
140    #    unittomicron
141    heightperpix = np.zeros(3)
142    for k in range(0, 3):
143        infostring = extract_tag(filecontent, b''.join([keywords[0], axes[k]]))
144        calibstring = extract_tag(filecontent, b''.join([keywords[6], axes[k]]))
145        print('---\n tomicron[k]: ', tomicron[k])
146        heightperpix[k] = str2float(infostring) * \
147                          (1e-6 * str2float(calibstring)) * tomicron[k]
148    #
149    endheightmap = filecontent.find(keywords[1])
150    #
151    infostring = extract_tag(filecontent, keywords[2])
152    magnification = str2float(infostring)
153    #
154    infostring = extract_tag(filecontent, keywords[4])
155    nx = int(infostring)
156    infostring = extract_tag(filecontent, keywords[5])
157    ny = int(infostring)
158    #
159    i1 = filecontent.find(keywords[3])
160    measdate = filecontent[i1 + len(keywords[3]) + 2:i1 + len(keywords[3]) + 22].decode('ascii')
161    return nx, ny, endheightmap, heightperpix, magnification, measdate

218def read_plu(filecontent):
219    DATE_SIZE = 128
220    COMMENT_SIZE = 256
221    measdate = filecontent[0:DATE_SIZE].decode('ascii', errors='ignore').replace("\x00", "")
222    print('--plu measdate: ', measdate)
223    i1 = DATE_SIZE + COMMENT_SIZE + 4
224    npix = struct.unpack('2I', filecontent[i1:i1 + 8])
225    pix_size = struct.unpack('2f', filecontent[i1 + 16:i1 + 24])
226    i2 = i1 + 10 * 4 + 10 * 4 + 8 + 3 * 4 + 1 + 7 + 4
227    num_height = npix[0] * npix[1]
228    data_height = struct.unpack(str(num_height) + 'f', filecontent[i2:i2 + 4 * num_height])
229    height_map = np.reshape(data_height, (npix[0], npix[1]))
230    inan = np.where(height_map > 1e6)
231    height_map[inan] = np.NaN
232    dx = pix_size[1]
233    dy = pix_size[0]
234    return npix[1], npix[0], dx, dy, height_map, measdate
